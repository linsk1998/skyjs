创建一个颜色对象

[new] Color("#FFFFFF");
[new] Color(0xffffff);

[new] Color("red");

对象方法

var red=new Color("red");
var blue=new Color("blue");
//判断颜色是否相同
red.equals(blue);//false
//转为HSL
red.toHSL();//[0, 1, 0.5]
//转为RGB
red.toRGB();//[255, 0, 0]
//转为获取值
red.valueOf();//16711680
//转为css使用的颜色代码
red.toHexString();//"#FF0000"
red.toRGBAString();//"rgba( 255, 0, 0, 1)"
red.toString();//#FF0000
//变亮，生成新对象
red.lighten(0.1);//Color(0xff3333)
//变暗，生成新对象
red.darken(0.1);//Color(0xcc0000)
//色轮，生成新对象
red.spin(180);//Color(0x00ffff)
//混合两种颜色
red.mix(blue,0.5);//Color(0x800080)
