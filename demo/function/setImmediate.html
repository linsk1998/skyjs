setImmediate

在浏览器完成后面的其他语句后,就立刻执行这个回调函数。多层setImmediate在setTimeout 0之前执行。

setTimeout(function(){alert(2)},0);
setImmediate(function(){
	setImmediate(function(){
		setImmediate(function(){
			setImmediate(function(){
				setImmediate(function(){
					setImmediate(function(){
						setImmediate(function(){
							alert(1);
						});
					});
				});
			});
		});
	});
});
setImmediate(function(){
	alert(0);
});
