数组操作

Sky.findIndex

查找对象数组编号

var users = [
	{ 'user': 'barney' ,'age':10 },
	{ 'user': 'fred','age':11 }
];

Sky.findIndex(users,"user","fred");
// => 1
Sky.findIndex(users,"user","barney");
// => 0
Sky.findIndex(users,"user","tom");
// => -1

Sky.findLastIndex

该方法类似findIndex ，区别是其从右到左遍历数组。

var users = [
	{ 'user': 'barney' ,'age':10 },
	{ 'user': 'fred','age':11 },
	{ 'user': 'fred','age':13 }
];

Sky.findLastIndex(users,"user","fred");
// => 2

Sky.find

查找对象数组中的对象。

var users = [
	{ 'user': 'barney' ,'age':10 },
	{ 'user': 'fred','age':11 }
];

Sky.find(users,"user","fred");
// => { 'user': 'fred','age':11 }

Sky.findLast

该方法类似find ，区别是其从右到左遍历数组。

var users = [
	{ 'user': 'barney' ,'age':10 },
	{ 'user': 'fred','age':11 },
	{ 'user': 'fred','age':13 }
];

Sky.findLast(users,"user","fred");
// => { 'user': 'fred','age':13 }

Sky.sortBy

按照对象数组中的字段排序。

var users = [
	{ 'user': 'pebbles','age':11 },
	{ 'user': 'barney' ,'age':10 },
	{ 'user': 'fred','age':13 }
];

Sky.sortBy(users, 'user');

Sky.pluck

对象数组转化为普通数组。

var users = [
	{ 'user': 'pebbles','age':11 },
	{ 'user': 'barney' ,'age':10 },
	{ 'user': 'fred','age':13 }
];

Sky.pluck(users, 'user');
// => ['pebbles','barney','fred']

Sky.sortedIndex

返回value 应该插入数组array位置的索引值

Sky.sortedIndex([30, 50], 40);
// => 1

Sky.sortedIndex([4, 4, 5, 5], 5);
// => 2

Sky.sortedLastIndex

用法类似于sortedIndex ,不同的是从右至左计算插入的位置

Sky.sortedLastIndex([4, 4, 5, 5], 5);
// => 4
