String操作

Sky.pad

开头补0

Sky.pad(5,4)

Sky.escape

对 HTML 中的特殊字符进行转义。

Sky.unescape

对HTML转义字符串转换成字符串。

document.getElementById("EHW").innerHTML=Sky.escape(prompt("请输入要转化的字符","<b>测试</b>"))

Sky.escapeString

对 JavaScript 中的字符串特殊字符进行转义。

alert(eval('"'+Sky.escapeString(prompt("请输入要转化的字符","\\t\\\""))+'"'))

Sky.escapeRegExp

对 JavaScript 中的正则表达式特殊字符进行转义。

alert("\\n\d\\n".replace(new RegExp(Sky.escapeRegExp("\d"), "g"), "HHH"))

Sky.strlen

获取字符串占位长度

alert(Sky.strlen("中文为2"))

Sky.trunc

截取字符串占位长度

alert(Sky.trunc("中文为2",5,"..."))

Sky.parseQuery

解析url参数

var param=Sky.parseQuery(location.search);
var param2=Sky.parseQuery("a=1&b=2&c=3");   //{"a":"1","b":"2","c":"3"}

Sky.buildQuery

转变为url数据(utf-8)

var param={"a":"1","b":"2","c":"3"};
alert(Sky.buildQuery(param));   //a=1&b=2&c=3

Sky.parseURL

解析URL

alert(JSON.stringify(Sky.parseURL(prompt("请输入URL地址","http://www.sss.ss:8080/asd/asd.asp?s=aa&ss=1#asd"))))

Sky.getAbsPath

获取绝对路径

alert(Sky.getAbsPath("../tet.html","http://www.sss.ss:8080/asd/asd.asp?s=aa&ss=1#asd"))
