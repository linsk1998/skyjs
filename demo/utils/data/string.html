String操作

Sky.pad

数字开头补0，字符串补全使用String.prototype.padStart。

Sky.pad(5,4)

Sky.unescape

对HTML转义字符串转换成字符串。

alert(Sky.unescape("&lt;b&gt;测试&lt;/b&gt;"))
运行

Sky.escapeHtml

对 HTML 中的特殊字符进行转义。

document.getElementById("EHW").innerHTML=Sky.escapeHtml(prompt("请输入要转化的字符","<b>测试</b>"));
运行

Sky.escapeString

对 JavaScript 中的字符串特殊字符进行转义。

alert(eval('"'+Sky.escapeString(prompt("请输入要转化的字符","\\t\\\""))+'"'))
运行

Sky.escapeRegExp

对 JavaScript 中的正则表达式特殊字符进行转义。

alert("\\n\d\\n".replace(new RegExp(Sky.escapeRegExp("\d"), "g"), "HHH"))
运行

Sky.replaceAll

替换多个子字符串。

alert(Sky.replaceAll("aaa/ssss/ddd/ff", "/", "-"))
运行

Sky.strlen

获取字符串占位长度,中文为2

alert(Sky.strlen("中文为2"))
运行

Sky.trunc

截取字符串占位长度

alert(Sky.trunc("中文为2",5,"..."))
运行

Sky.buildQuery

转变为url数据(utf-8)

var param={"a":"1","b":"2","c":"3"};
alert(Sky.buildQuery(param));   //a=1&b=2&c=3
运行

Sky.parseQuery

解析url参数

var param=Sky.parseQuery(location.search);
var param2=Sky.parseQuery("a=1&b=2&c=3");   //{"a":"1","b":"2","c":"3"}
alert(JSON.stringify(param2));
运行
