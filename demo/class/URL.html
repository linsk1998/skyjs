URL

对象保存键值对。任何值(对象或者原始值) 都可以作为一个键或一个值。

属性描述例子
href完整 URLhttp://shengdiyage.us:80/abc/index.asp?id=1&t=1400#top
protocol协议名，末尾带 ':'http:
host域名，':'，和端口号shengdiyage.us:80
hostname域名shengdiyage.us
port端口号80
pathname以 '/' 起头紧跟着 URL 文件路径/abc/index.asp
search以 '?' 起头紧跟着 URL 请求参数?id=1&t=1400
hash以 '#' 起头紧跟着 URL 锚点标记#top
origin返回一个包含协议名、域名和端口号http://shengdiyage.us:80
searchParams返回一个用来访问当前 URL GET 请求参数的 URLSearchParams 对象 
//解析URL
var url=new URL("http://shengdiyage.us:80/abc/index.asp?id=1&t=1400#top");
var a=[];
for(var key in url){
	a.push(key+":"+url[key]);
}
alert(a.join("\n"))
alert("id="+url.searchParams.get("id"))
运行 
//获取绝对路径
alert(new URL("/index.html",location).href);
alert(new URL("?name=admin",location).href);
alert(new URL("#/eg",location).href);
alert(new URL("./index.aspx",location).href);
alert(new URL("../../index.php",location).href);
运行 
//修改属性
var url=new URL("http://shengdiyage.us/index.asp?id=1&t=1400");
url.searchParams.set('id',"2");
url.hostname="linsk.me";
alert(url.href);
