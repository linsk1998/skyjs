Set

无序集合

方法返回描述
add添加元素
has返回布尔值判断是否存在某个元素
delete删除某个元素
clear清除内部所有元素
forEach对数组的每个元素执行一次提供的函数(回调函数) 
//基本用法
var s = new Set();
s.add("Thomas Jefferson");
s.add(1776);
s.add("founding father");

s.forEach(function (item) {
	alert(item);
});
运行 
//删除
var s = new Set([1,2,3,3]);
s['delete'](3);
alert(s.size);//2
s.clear();
alert(s.size);//0
运行 
//特殊类型
var s = new Set([1,"1",new String("1"),new Number(1),NaN,0,false,true,null,undefined]);
alert(s.size);//10

s.forEach(function (item) {
	alert(item);
});
运行 
//转数组
var s=new Set([1,2,3,3]);
alert(Array.from(s).join(","));
运行
