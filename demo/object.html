对象操作

Sky.applyIf

属性拷贝不会覆盖

Sky.extend

扩展对象，一般推荐使用Object.assign。

var result=$.extend({},{name:"Tom",age:21},{name:"Jerry",sex:"Boy"});
//{name: "Jerry", age: 21, sex: "Boy"}

第一个参数boolean代表是否进行深度拷贝

var result=$.extend( true, {}, { name: "John", location: {city: "Boston",county:"USA"} }, { last: "Resig", location: {state: "MA",county:"China"} } );
//{"name":"John","location":{"city":"Boston","county":"China","state":"MA"},"last":"Resig"}

Sky.forIn forOwn

遍历对象的元素。

Sky.forIn( {"key":"value"}, function( value, key){
	alert(this[key]);
});

Sky.parseQuery

解析url参数

var param=Sky.parseQuery(location.search);
var param2=Sky.parseQuery("a=1&b=2&c=3");   //{"a":"1","b":"2","c":"3"}

Sky.buildQuery

转变为url数据(utf-8)

var param={"a":"1","b":"2","c":"3"};
alert(Sky.buildQuery(param));   //a=1&b=2&c=3

Sky.isXxx

判断类型

方法描述
isArray是否是数组
isDate是否是日期
isString是否是字符串
isRegExp是否是正则表达式
isFunction是否是函数
isNumber是否是数字
isObject是否是对象（typeof）
isPlainObject是否是纯对象
isNumeric是否是数字 "10" "10.3"
isDefined是否未定义
isArrayLike是否是类数组
isElement是否是元素
isEmpty是否是空 包括Set Array ArrayLike
